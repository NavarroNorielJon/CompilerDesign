<operator> -> \+|\*|\-|div|mod
<log_operator> -> <|>|!=|==|<=|>=|not
<rel_operator> -> and|or
<assign> -> is
<quot_string> -> "
<quot_char> -> '
<print> -> print
<end> -> end
<datatype> -> integer|decimal|long|boolean|character|string
<if> -> if
<end> -> end
<until> -> until
<class> -> class
<access_mod> -> public|private
<bool_val> -> true|false
<scanner> -> scanner
<number> -> -?[0-9]+
<iden> -> [a-z][a-z0-9_]*
