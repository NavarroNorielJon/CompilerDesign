<assign> -> is
<class> -> class
<end> -> \.
<end_if> -> end if
<end_program> -> end program
<end_while> -> end while
<if> -> if
<print> -> print
<scanner> -> scanner
<until> -> until
<access_modifier> -> public|private
<data_type> -> integer|decimal|long|boolean|character|string
<logical_operator> -> <|>|!=|==|<=|>=|not
<operator> -> \+|\*|\-|div|mod
<relational_operator> -> and|or
<boolean_value> -> true|false
<character> -> '[a-zA-z0-9]'
<program_name> -> [A-Z][a-zA-Z0-9]+
<identifier> -> [a-z][a-zA-Z0-9_]+
<string> -> "[a-zA-Z0-9_ ]*"
<number> -> -?[0-9]+
