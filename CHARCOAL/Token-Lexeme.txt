

<operator> -> \+ | \* | \- |div | mod
<log_operator> -> < | > | != | == | <= | >= | not
<rel_operator> -> and | or
<assign> -> is
<quot_string> -> "
<quot_char> -> '
<print> -> print
<number> -> -?[0-9]+
<end> -> end
<datatype> -> integer | decimal | long | boolean | character | string
<if> -> if
<else> -> else
<end> -> end
<until> -> until
<class> -> class
<access_mod> public | private
<bool_val> -> true | false
<scanner> -> scanner
<iden> -> [a-z][a-z0-9_]
